#Clean work place 
module purge

#Charger les modules necessaires 
module load bioinfo/bwa-mem2/2.2.1
module load bioinfo/samtools/1.19

# Move to the directorie containing files
cd ~/work/Dicorynia/Raw_data/ 

#Indexing the referance genome
#bwa-mem2 index Dgu_HS1_HYBRID_SCAFFOLD.fa
#reference genome already indexed 

#Creating a variable pointing to the reference genome and the individual 
REF=~/work/Dicorynia/Raw_data/Dgu_HS1_HYBRID_SCAFFOLD.fa


INDS=($(for i in ~/work/Dicorynia/test_subsamples/raw_subsamples/*_clean_subsampled.fastq; do
    base=$(basename "$i")
    echo "${base%%.R[12]_clean_subsampled.fastq}" 
done))


#to check if INDS work 
echo "${INDS[@]}"


#Declare variables separate foward and reverse 


for IND in ${INDS[@]};
do
  FORWARD=~/work/Dicorynia/test_subsamples/raw_subsamples/${IND}.R1_clean_subsampled.fastq
  REVERSE=~/work/Dicorynia/test_subsamples/raw_subsamples/${IND}.R2_clean_subsampled.fastq
  OUTPUT=~/work/Dicorynia/test_subsamples/align_subsamples/${IND}_sort.bam

  echo "${FORWARD[@]}"
  echo "${REVERSE[@]}"
  

  # then align and sort
  echo "Aligning $IND with bwa"
  bwa-mem2 mem -M -t 4 $REF $FORWARD $REVERSE | samtools view -b | samtools sort -T ${IND} > $OUTPUT

done
