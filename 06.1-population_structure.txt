##### Population structure analysis #####

echo "Starting Population structure analysis"

# Cleaning work place 
module purge

# Load the necessary modules
module load bioinfo/PLINK/2.00a4 
module load bioinfo/ADMIXTURE/1.3.0

# Define paths
BINARY_FILES_PREFIX="/work/user/jbonnier/Dicorynia/full_workflow/05-filtering_variants/05.3-missing_filtering/missing_filtered"
ADMIXTURE_OUTPUT="/work/user/jbonnier/Dicorynia/full_workflow/06-population_structure/06.1-admixture/"

# Renaming chromosomes to numbers in .bim file to make it compatible with ADMIXTURE
echo "Renaming chromosomes in .bim file..."
awk '{$1="0"; print}' ${BINARY_FILES_PREFIX}.bim > ${BINARY_FILES_PREFIX}_renamed.bim
mv ${BINARY_FILES_PREFIX}_renamed.bim ${BINARY_FILES_PREFIX}.bim

# Run ADMIXTURE for a range of K values
echo "Running ADMIXTURE..."

for K in {1..15}; do
    admixture --cv ${BINARY_FILES_PREFIX}.bed $K | tee ${ADMIXTURE_OUTPUT}/log${K}.out
done

echo "End of ADMIXTURE processing, let's see the results"
