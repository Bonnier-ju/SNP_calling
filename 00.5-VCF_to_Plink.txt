#SBATCH -p workq
#SBATCH --time=20:00:00 #job time limit
#SBATCH -J 00.5-VCF_to_Plink #job name
#SBATCH -o /work/user/jbonnier/Dicorynia/full_workflow/00-batch_scripts/output/output-%x-%j.out #output file name

echo "Starting conversion to binary file"

# Clean the work environment
module purge

#Module load
module load bioinfo/PLINK/1.90b7


# Define file paths
VCF_FILE="/work/user/jbonnier/Dicorynia/full_workflow/05-filtering_variants/05.6-subsampling_SNP/sub_200k_SNP/sub_200k_SNP_full/sub_200k_SNP_full.vcf"
OUTPUT="/work/user/jbonnier/Dicorynia/full_workflow/05-filtering_variants/05.6-subsampling_SNP/sub_200k_SNP/sub_200k_SNP_full/sub_200k_SNP_full"
EXCLUDE="/work/user/jbonnier/Dicorynia/full_workflow/06-population_structure/exclude_samples_MC87_Angela.txt"



plink --vcf $VCF_FILE \
	--make-bed \
  --exclude $EXCLUDE \
  --recode \
  --const-fid \
	--allow-extra-chr \
	--out $OUTPUT
